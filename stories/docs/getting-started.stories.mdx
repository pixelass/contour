# Getting Started

## Install

```shell
# npm
npm install @contour/react --save;
# yarn
yarn add @contour/react --save
```

## Basic Usage

```jsx
import { Column, Grid } from "@contour/react";

export default function App() {
	return (
		<Grid>
			<Column>Contour</Column>
			<Column colSpan={{ xs: 2 }}>Contour</Column>
			<Column colSpan={{ xs: 2, m: 4 }}>Contour</Column>
		</Grid>
	);
}
```
