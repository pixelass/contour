# Getting Started

## Install

```shell
# npm
npm install @contour/react --save;
# yarn
yarn add @contour/react --save
```

## Basic Usage

```tsx
import Grid from "@contour/react/grid";
import Column from "@contour/react/column";

export default function App() {
	return (
		<Grid>
			<Column>Contour</Column>
			<Column colSpan={{ xs: 2 }}>Contour</Column>
			<Column colSpan={{ xs: 2 }}>Contour</Column>
		</Grid>
	);
}
```
